async function calcularRota() { if(!minhaLat || !minhaLon) { document.getElementById('rua-atual').innerText = "AGUARDANDO SINAL DE GPS..."; return; } const d = rotaSpx[idxDestino]; document.getElementById('stop-badge').innerText = "STOP " + d.stop; if (markerDest) mapa.removeLayer(markerDest); markerDest = L.circleMarker([d.lat, d.lon], { radius: 11, color: '#fff', fillColor: '#007AFF', fillOpacity: 1, weight: 3 }).addTo(mapa); // LOG PARA VOCÊ VER NO CONSOLE F12 console.log("Tentando rota para:", d.lat, d.lon, "Minha Pos:", minhaLat, minhaLon); try { const url = `https://router.project-osrm.org/route/v1/driving/${minhaLon},${minhaLat};${d.lon},${d.lat}?steps=true&overview=full&geometries=geojson`; const res = await fetch(url); if (!res.ok) { // Se a API responder, mas com erro (ex: 429 limite de uso ou 400 erro de rota) const errorText = await res.text(); document.getElementById('rua-atual').innerText = "ERRO API: " + res.status + " - " + res.statusText; console.error("Detalhe do Erro:", errorText); return; } const data = await res.json(); if (data.routes && data.routes[0]) { const route = data.routes[0]; passosNavegacao = route.legs[0].steps.map(s => ({ lat: s.maneuver.location[1], lon: s.maneuver.location[0], manobra: s.maneuver.modifier, rua: s.name || "Siga em frente" })); idxPasso = 0; if (polylineRota) mapa.removeLayer(polylineRota); polylineRota = L.polyline(route.geometry.coordinates.map(c => [c[1], c[0]]), { color: '#007AFF', weight: 6 }).addTo(mapa); document.getElementById('rua-atual').innerText = passosNavegacao[0].rua; } else { document.getElementById('rua-atual').innerText = "ROTA NÃO ENCONTRADA NESTE PONTO"; } } catch (e) { // Erro de rede (falta de internet ou servidor caiu totalmente) document.getElementById('rua-atual').innerText = "ERRO DE REDE/CONEXÃO: " + e.message; } }
